cmake_minimum_required(VERSION 3.20)

project(STM32L4_RTOS C ASM)

set(LINKER_SCRIPT "${CMAKE_SOURCE_DIR}/STM32L432KCUX_FLASH.ld")

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

add_subdirectory(FreeRTOS)
add_subdirectory(Drivers)
add_subdirectory(Lib)

add_executable(${PROJECT_NAME}
    ${CMAKE_SOURCE_DIR}/Core/main/main.c
    ${CMAKE_SOURCE_DIR}/Core/main/init.c
    ${CMAKE_SOURCE_DIR}/Core/main/stm32l4xx_hal_msp.c
    ${CMAKE_SOURCE_DIR}/Core/main/stm32l4xx_it.c
    ${CMAKE_SOURCE_DIR}/Core/main/callbacks.c
    ${CMAKE_SOURCE_DIR}/Core/main/syscalls.c
    ${CMAKE_SOURCE_DIR}/Core/main/sysmem.c

    ${CMAKE_SOURCE_DIR}/Core/mpu6050/imu_data_input.c

    ${CMAKE_SOURCE_DIR}/Core/nrf905/driver_nrf905_interface.c
    ${CMAKE_SOURCE_DIR}/Core/nrf905/nrf905_init.c
    ${CMAKE_SOURCE_DIR}/Core/nrf905/nrf905_link.c

    ${CMAKE_SOURCE_DIR}/Core/Startup/startup_stm32l432xx.s
)

target_link_options(${PROJECT_NAME} PRIVATE
    -Wl,-Map=${PROJECT_NAME}.map
    -T${LINKER_SCRIPT}
)

target_link_libraries(${PROJECT_NAME} PRIVATE
    freertos
    hal
    mpu6050_lib
    nrf905_lib
)

target_include_directories(${PROJECT_NAME}
    PRIVATE
        ${CMAKE_SOURCE_DIR}/Core/main/inc
        ${CMAKE_SOURCE_DIR}/Core/mpu6050/inc
        ${CMAKE_SOURCE_DIR}/Core/nrf905/inc
)

add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary $<TARGET_FILE:${PROJECT_NAME}> ${PROJECT_NAME}.bin
    COMMAND ${CMAKE_OBJCOPY} -O ihex   $<TARGET_FILE:${PROJECT_NAME}> ${PROJECT_NAME}.hex
    COMMAND ${CMAKE_SIZE} -B $<TARGET_FILE:${PROJECT_NAME}>
)
